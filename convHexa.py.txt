def convHex(resto):
    resultado = str(resto)
    if resto == 10:
        resultado = 'a'
    elif resto == 11:
        resultado = 'b'
    elif resto == 12:
        resultado = 'c'
    elif resto == 13:
        resultado = 'd'
    elif resto == 14:
        resultado = 'e'
    elif resto == 15:
        resultado = 'f'
    return resultado    
base=int(input('Defina a base numerica: '))
valorbase = ''
numero=int(input('Digite o Numero: '))
while numero > 1:
    resto = numero % base
    print(convHex(resto))
    valorbase = convHex(resto) + valorbase
    numero = numero // base
    if numero == 0 or numero == 1:
        valorbase = str(numero) + valorbase
print (valorbase)
